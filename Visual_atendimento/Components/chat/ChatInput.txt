import React, { useState, useRef, useEffect } from 'react';
import { Button } from "@/components/ui/button";
import { Textarea } from "@/components/ui/textarea";
import { 
  Send, 
  Paperclip, 
  Package,
  X,
  Loader2
} from "lucide-react";
import {
  Popover,
  PopoverContent,
  PopoverTrigger,
} from "@/components/ui/popover";

export default function ChatInput({ 
  value,
  onChange,
  onSend, 
  onAttach, 
  onSendProduct,
  disabled,
  placeholder = "Digite sua mensagem..." 
}) {
  const [message, setMessage] = useState(value || '');
  const [attachments, setAttachments] = useState([]);
  const [isSending, setIsSending] = useState(false);
  const textareaRef = useRef(null);
  const fileInputRef = useRef(null);

  // Sync external value
  useEffect(() => {
    if (value !== undefined && value !== message) {
      setMessage(value);
    }
  }, [value]);

  const handleSend = async () => {
    if (!message.trim() && attachments.length === 0) return;
    setIsSending(true);
    await onSend(message, attachments);
    setMessage('');
    setAttachments([]);
    if (onChange) onChange('');
    setIsSending(false);
    textareaRef.current?.focus();
  };

  const handleChange = (e) => {
    const newValue = e.target.value;
    setMessage(newValue);
    if (onChange) onChange(newValue);
  };

  const handleKeyDown = (e) => {
    if (e.key === 'Enter' && !e.shiftKey) {
      e.preventDefault();
      handleSend();
    }
  };

  const handleFileSelect = (e) => {
    const files = Array.from(e.target.files || []);
    setAttachments(prev => [...prev, ...files]);
  };

  const removeAttachment = (index) => {
    setAttachments(prev => prev.filter((_, i) => i !== index));
  };

  return (
    <div className="flex-1">
      {attachments.length > 0 && (
        <div className="flex gap-2 mb-2 flex-wrap">
          {attachments.map((file, index) => (
            <div 
              key={index}
              className="flex items-center gap-2 px-3 py-1.5 bg-slate-100 rounded-lg"
            >
              <span className="text-sm text-slate-600 truncate max-w-[150px]">
                {file.name}
              </span>
              <button 
                onClick={() => removeAttachment(index)}
                className="text-slate-400 hover:text-slate-600"
              >
                <X className="w-4 h-4" />
              </button>
            </div>
          ))}
        </div>
      )}
      
      <div className="flex items-end gap-2">
        <div className="flex gap-1">
          <input
            ref={fileInputRef}
            type="file"
            multiple
            className="hidden"
            onChange={handleFileSelect}
          />
          
          <Button
            variant="ghost"
            size="icon"
            className="text-slate-400 hover:text-slate-600 hover:bg-slate-100"
            onClick={() => fileInputRef.current?.click()}
            disabled={disabled}
          >
            <Paperclip className="w-5 h-5" />
          </Button>
          
          {onSendProduct && (
            <Popover>
              <PopoverTrigger asChild>
                <Button
                  variant="ghost"
                  size="icon"
                  className="text-slate-400 hover:text-slate-600 hover:bg-slate-100"
                  disabled={disabled}
                >
                  <Package className="w-5 h-5" />
                </Button>
              </PopoverTrigger>
              <PopoverContent className="w-64 p-2" align="start">
                <p className="text-sm text-slate-500 p-2">
                  Enviar produto do cat√°logo
                </p>
                <Button 
                  variant="outline" 
                  className="w-full justify-start"
                  onClick={onSendProduct}
                >
                  <Package className="w-4 h-4 mr-2" />
                  Selecionar produto
                </Button>
              </PopoverContent>
            </Popover>
          )}
        </div>
        
        <div className="flex-1 relative">
          <Textarea
            ref={textareaRef}
            value={message}
            onChange={handleChange}
            onKeyDown={handleKeyDown}
            placeholder={placeholder}
            disabled={disabled}
            className="min-h-[44px] max-h-[120px] resize-none rounded-2xl border-slate-200 focus:border-slate-300 focus:ring-slate-200"
            rows={1}
          />
        </div>
        
        <Button
          onClick={handleSend}
          disabled={disabled || isSending || (!message.trim() && attachments.length === 0)}
          className="h-11 w-11 rounded-full bg-slate-800 hover:bg-slate-700 p-0"
        >
          {isSending ? (
            <Loader2 className="w-5 h-5 animate-spin" />
          ) : (
            <Send className="w-5 h-5" />
          )}
        </Button>
      </div>
    </div>
  );
}
