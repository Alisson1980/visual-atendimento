import React, { useState } from 'react';
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { base44 } from '@/api/base44Client';
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Badge } from "@/components/ui/badge";
import { ScrollArea } from "@/components/ui/scroll-area";
import {
  Popover,
  PopoverContent,
  PopoverTrigger,
} from "@/components/ui/popover";
import { 
  BookOpen, 
  Search, 
  Copy, 
  Check,
  ChevronRight,
  FileText,
  Package,
  DollarSign,
  Truck,
  CreditCard,
  Shield,
  HelpCircle,
  FileCode
} from "lucide-react";

const categoryConfig = {
  products: { label: 'Produtos', icon: Package, color: 'bg-blue-100 text-blue-700' },
  processes: { label: 'Processos', icon: FileCode, color: 'bg-purple-100 text-purple-700' },
  pricing: { label: 'Preços', icon: DollarSign, color: 'bg-green-100 text-green-700' },
  shipping: { label: 'Envio', icon: Truck, color: 'bg-orange-100 text-orange-700' },
  payment: { label: 'Pagamento', icon: CreditCard, color: 'bg-emerald-100 text-emerald-700' },
  policies: { label: 'Políticas', icon: Shield, color: 'bg-slate-100 text-slate-700' },
  faq: { label: 'FAQ', icon: HelpCircle, color: 'bg-amber-100 text-amber-700' },
  templates: { label: 'Templates', icon: FileText, color: 'bg-violet-100 text-violet-700' }
};

export default function KnowledgeBasePanel({ onInsert }) {
  const queryClient = useQueryClient();
  const [open, setOpen] = useState(false);
  const [search, setSearch] = useState('');
  const [selectedCategory, setSelectedCategory] = useState(null);
  const [copiedId, setCopiedId] = useState(null);

  const { data: articles = [] } = useQuery({
    queryKey: ['knowledge-base'],
    queryFn: () => base44.entities.KnowledgeBase.filter({ is_active: true }, '-use_count', 100),
    enabled: open
  });

  const updateUsage = useMutation({
    mutationFn: (id) => {
      const article = articles.find(a => a.id === id);
      return base44.entities.KnowledgeBase.update(id, { 
        use_count: (article?.use_count || 0) + 1 
      });
    }
  });

  const filteredArticles = articles.filter(article => {
    const matchesSearch = !search || 
      article.title.toLowerCase().includes(search.toLowerCase()) ||
      article.content.toLowerCase().includes(search.toLowerCase()) ||
      article.tags?.some(t => t.toLowerCase().includes(search.toLowerCase()));
    
    const matchesCategory = !selectedCategory || article.category === selectedCategory;
    
    return matchesSearch && matchesCategory;
  });

  const handleInsert = (article) => {
    const textToInsert = article.short_answer || article.content;
    onInsert(textToInsert);
    updateUsage.mutate(article.id);
    setOpen(false);
    setSearch('');
  };

  const handleCopy = (article) => {
    const textToCopy = article.short_answer || article.content;
    navigator.clipboard.writeText(textToCopy);
    setCopiedId(article.id);
    updateUsage.mutate(article.id);
    setTimeout(() => setCopiedId(null), 2000);
  };

  return (
    <Popover open={open} onOpenChange={setOpen}>
      <PopoverTrigger asChild>
        <Button 
          variant="ghost" 
          size="icon" 
          className="text-blue-600 hover:text-blue-700 hover:bg-blue-50"
        >
          <BookOpen className="w-5 h-5" />
        </Button>
      </PopoverTrigger>
      <PopoverContent className="w-96 p-0" align="start" side="top">
        <div className="p-3 border-b">
          <div className="flex items-center gap-2 mb-3">
            <BookOpen className="w-5 h-5 text-blue-600" />
            <h3 className="font-semibold">Base de Conhecimento</h3>
          </div>
          <div className="relative">
            <Search className="absolute left-2.5 top-1/2 -translate-y-1/2 w-4 h-4 text-slate-400" />
            <Input
              value={search}
              onChange={(e) => setSearch(e.target.value)}
              placeholder="Buscar artigos, FAQ..."
              className="pl-9 h-9"
            />
          </div>
        </div>

        {/* Categories */}
        <div className="p-2 border-b flex flex-wrap gap-1">
          <Button
            size="sm"
            variant={!selectedCategory ? "default" : "ghost"}
            className="h-7 text-xs"
            onClick={() => setSelectedCategory(null)}
          >
            Todos
          </Button>
          {Object.entries(categoryConfig).map(([key, config]) => {
            const Icon = config.icon;
            return (
              <Button
                key={key}
                size="sm"
                variant={selectedCategory === key ? "default" : "ghost"}
                className="h-7 text-xs"
                onClick={() => setSelectedCategory(key)}
              >
                <Icon className="w-3 h-3 mr-1" />
                {config.label}
              </Button>
            );
          })}
        </div>

        {/* Articles */}
        <ScrollArea className="h-64">
          <div className="p-2 space-y-1">
            {filteredArticles.length === 0 ? (
              <div className="text-center py-8 text-slate-500">
                <HelpCircle className="w-8 h-8 mx-auto mb-2 opacity-50" />
                <p className="text-sm">Nenhum artigo encontrado</p>
              </div>
            ) : (
              filteredArticles.map(article => {
                const config = categoryConfig[article.category];
                const Icon = config?.icon || FileText;
                return (
                  <div
                    key={article.id}
                    className="p-2 rounded-lg hover:bg-slate-50 group"
                  >
                    <div className="flex items-start gap-2">
                      <div className={`p-1.5 rounded ${config?.color || 'bg-slate-100'}`}>
                        <Icon className="w-3.5 h-3.5" />
                      </div>
                      <div className="flex-1 min-w-0">
                        <p className="font-medium text-sm text-slate-900 truncate">
                          {article.title}
                        </p>
                        <p className="text-xs text-slate-500 line-clamp-2 mt-0.5">
                          {article.short_answer || article.content}
                        </p>
                        {article.tags?.length > 0 && (
                          <div className="flex gap-1 mt-1 flex-wrap">
                            {article.tags.slice(0, 3).map((tag, idx) => (
                              <Badge key={idx} variant="outline" className="text-xs px-1 py-0">
                                {tag}
                              </Badge>
                            ))}
                          </div>
                        )}
                      </div>
                      <div className="flex gap-1 opacity-0 group-hover:opacity-100 transition-opacity">
                        <Button
                          size="sm"
                          variant="ghost"
                          className="h-7 w-7 p-0"
                          onClick={() => handleCopy(article)}
                        >
                          {copiedId === article.id ? (
                            <Check className="w-3.5 h-3.5 text-green-600" />
                          ) : (
                            <Copy className="w-3.5 h-3.5" />
                          )}
                        </Button>
                        <Button
                          size="sm"
                          className="h-7 text-xs"
                          onClick={() => handleInsert(article)}
                        >
                          Inserir
                          <ChevronRight className="w-3 h-3 ml-1" />
                        </Button>
                      </div>
                    </div>
                  </div>
                );
              })
            )}
          </div>
        </ScrollArea>
      </PopoverContent>
    </Popover>
  );
}
