import React from 'react';
import { cn } from "@/lib/utils";
import { format, isToday, isYesterday } from "date-fns";
import { ptBR } from "date-fns/locale";
import { Badge } from "@/components/ui/badge";
import { Bot, User, Clock, MessageCircle } from "lucide-react";
import { ChannelBadge, ChannelIcon } from './ChannelBadge';

const statusColors = {
  waiting: "bg-amber-100 text-amber-700 border-amber-200",
  ai_handling: "bg-violet-100 text-violet-700 border-violet-200",
  human_handling: "bg-blue-100 text-blue-700 border-blue-200",
  pending_transfer: "bg-orange-100 text-orange-700 border-orange-200",
  resolved: "bg-emerald-100 text-emerald-700 border-emerald-200",
  archived: "bg-slate-100 text-slate-500 border-slate-200"
};

const statusLabels = {
  waiting: "Aguardando",
  ai_handling: "IA",
  human_handling: "Em atendimento",
  pending_transfer: "Na fila",
  resolved: "Resolvido",
  archived: "Arquivado"
};

const priorityIndicator = {
  urgent: "bg-red-500",
  high: "bg-orange-500",
  normal: "bg-transparent",
  low: "bg-transparent"
};

function formatMessageDate(dateString) {
  if (!dateString) return "";
  const date = new Date(dateString);
  if (isToday(date)) return format(date, "HH:mm");
  if (isYesterday(date)) return "Ontem";
  return format(date, "dd/MM", { locale: ptBR });
}

export default function ConversationList({ 
  conversations, 
  selectedId, 
  onSelect,
  isLoading 
}) {
  if (isLoading) {
    return (
      <div className="space-y-2 p-3">
        {[1, 2, 3, 4, 5].map(i => (
          <div key={i} className="animate-pulse">
            <div className="flex gap-3 p-3 rounded-xl bg-slate-50">
              <div className="w-12 h-12 rounded-full bg-slate-200" />
              <div className="flex-1 space-y-2">
                <div className="h-4 bg-slate-200 rounded w-2/3" />
                <div className="h-3 bg-slate-200 rounded w-full" />
              </div>
            </div>
          </div>
        ))}
      </div>
    );
  }

  if (!conversations?.length) {
    return (
      <div className="flex flex-col items-center justify-center h-64 text-slate-400">
        <MessageCircle className="w-12 h-12 mb-3 opacity-50" />
        <p className="text-sm">Nenhuma conversa</p>
      </div>
    );
  }

  return (
    <div className="space-y-1 p-2">
      {conversations.map(conv => (
        <button
          key={conv.id}
          onClick={() => onSelect(conv)}
          className={cn(
            "w-full flex gap-3 p-3 rounded-xl transition-all text-left",
            "hover:bg-slate-50",
            selectedId === conv.id 
              ? "bg-slate-100 ring-1 ring-slate-200" 
              : "bg-white"
          )}
        >
          <div className="relative flex-shrink-0">
            <div className={cn(
              "w-12 h-12 rounded-full flex items-center justify-center",
              conv.status === 'ai_handling' ? "bg-violet-100" : "bg-slate-100"
            )}>
              {conv.status === 'ai_handling' ? (
                <Bot className="w-5 h-5 text-violet-600" />
              ) : (
                <span className="text-lg font-semibold text-slate-600">
                  {conv.customer_name?.charAt(0)?.toUpperCase() || "?"}
                </span>
              )}
            </div>
            {priorityIndicator[conv.priority] !== "bg-transparent" && (
              <div className={cn(
                "absolute -top-0.5 -right-0.5 w-3 h-3 rounded-full border-2 border-white",
                priorityIndicator[conv.priority]
              )} />
            )}
          </div>
          
          <div className="flex-1 min-w-0">
            <div className="flex items-center justify-between mb-1">
              <span className="font-medium text-slate-900 truncate">
                {conv.customer_name || conv.customer_phone}
              </span>
              <span className="text-xs text-slate-400 flex-shrink-0 ml-2">
                {formatMessageDate(conv.last_message_at)}
              </span>
            </div>
            
            <p className="text-sm text-slate-500 truncate mb-2">
              {conv.last_message || "Nova conversa"}
            </p>
            
            <div className="flex items-center gap-2 flex-wrap">
              <ChannelBadge channel={conv.channel} showLabel={false} size="sm" />
              
              <Badge variant="outline" className={cn("text-xs px-2 py-0", statusColors[conv.status])}>
                {statusLabels[conv.status]}
              </Badge>
              
              {conv.unread_count > 0 && (
                <span className="px-2 py-0.5 bg-emerald-500 text-white text-xs font-medium rounded-full">
                  {conv.unread_count}
                </span>
              )}
              
              {conv.assigned_to_name && conv.status === 'human_handling' && (
                <span className="text-xs text-slate-400 truncate">
                  {conv.assigned_to_name}
                </span>
              )}
            </div>
          </div>
        </button>
      ))}
    </div>
  );
}
