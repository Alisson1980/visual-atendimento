import React, { useState } from 'react';
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { base44 } from '@/api/base44Client';
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Badge } from "@/components/ui/badge";
import { Label } from "@/components/ui/label";
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
  DialogDescription,
} from "@/components/ui/dialog";
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@/components/ui/select";
import { 
  Search, 
  Plus,
  Link as LinkIcon,
  Copy,
  Check,
  Users,
  Key,
  Clock,
  XCircle,
  RefreshCw,
  ExternalLink,
  Trash2
} from "lucide-react";
import { format } from "date-fns";
import { ptBR } from "date-fns/locale";

export default function ClientAccessManager() {
  const queryClient = useQueryClient();
  const [searchTerm, setSearchTerm] = useState('');
  const [showCreateModal, setShowCreateModal] = useState(false);
  const [selectedCustomerId, setSelectedCustomerId] = useState('');
  const [copiedId, setCopiedId] = useState(null);

  // Load accesses
  const { data: accesses = [], isLoading } = useQuery({
    queryKey: ['client-accesses'],
    queryFn: () => base44.entities.ClientAccess.list('-created_date', 200),
  });

  // Load customers
  const { data: customers = [] } = useQuery({
    queryKey: ['customers'],
    queryFn: () => base44.entities.Customer.list('name', 500),
  });

  // Create access
  const createAccess = useMutation({
    mutationFn: async (customerId) => {
      const customer = customers.find(c => c.id === customerId);
      if (!customer) throw new Error('Cliente não encontrado');

      // Check if access already exists
      const existing = accesses.find(a => a.customer_id === customerId && a.is_active);
      if (existing) throw new Error('Cliente já possui acesso ativo');

      const token = `${Date.now().toString(36)}-${Math.random().toString(36).substring(2, 15)}`;
      
      return base44.entities.ClientAccess.create({
        customer_id: customerId,
        customer_name: customer.name,
        customer_email: customer.email,
        customer_phone: customer.phone,
        access_token: token,
        is_active: true
      });
    },
    onSuccess: () => {
      queryClient.invalidateQueries(['client-accesses']);
      setShowCreateModal(false);
      setSelectedCustomerId('');
    }
  });

  // Toggle access
  const toggleAccess = useMutation({
    mutationFn: async ({ id, is_active }) => {
      return base44.entities.ClientAccess.update(id, { is_active });
    },
    onSuccess: () => {
      queryClient.invalidateQueries(['client-accesses']);
    }
  });

  // Regenerate token
  const regenerateToken = useMutation({
    mutationFn: async (id) => {
      const newToken = `${Date.now().toString(36)}-${Math.random().toString(36).substring(2, 15)}`;
      return base44.entities.ClientAccess.update(id, { access_token: newToken });
    },
    onSuccess: () => {
      queryClient.invalidateQueries(['client-accesses']);
    }
  });

  // Delete access
  const deleteAccess = useMutation({
    mutationFn: (id) => base44.entities.ClientAccess.delete(id),
    onSuccess: () => {
      queryClient.invalidateQueries(['client-accesses']);
    }
  });

  const getPortalUrl = (token) => {
    return `${window.location.origin}/ClientPortal?token=${token}`;
  };

  const copyToClipboard = async (text, id) => {
    await navigator.clipboard.writeText(text);
    setCopiedId(id);
    setTimeout(() => setCopiedId(null), 2000);
  };

  // Filter
  const filteredAccesses = accesses.filter(access => {
    return !searchTerm || 
      access.customer_name?.toLowerCase().includes(searchTerm.toLowerCase()) ||
      access.customer_email?.toLowerCase().includes(searchTerm.toLowerCase()) ||
      access.customer_phone?.includes(searchTerm);
  });

  // Customers without access
  const customersWithoutAccess = customers.filter(c => 
    !accesses.some(a => a.customer_id === c.id && a.is_active)
  );

  return (
    <div className="min-h-screen bg-slate-50 p-6">
      <div className="max-w-7xl mx-auto space-y-6">
        {/* Header */}
        <div className="flex flex-col md:flex-row md:items-center md:justify-between gap-4">
          <div>
            <h1 className="text-2xl font-bold text-slate-900">Acesso de Clientes</h1>
            <p className="text-slate-500">Gerencie os links de acesso ao portal do cliente</p>
          </div>
          <Button onClick={() => setShowCreateModal(true)} className="bg-slate-800 hover:bg-slate-700">
            <Plus className="w-4 h-4 mr-2" />
            Criar Acesso
          </Button>
        </div>

        {/* Stats */}
        <div className="grid grid-cols-2 md:grid-cols-3 gap-4">
          <Card className="border-0 shadow-sm">
            <CardContent className="p-4">
              <div className="flex items-center gap-3">
                <div className="p-2 bg-blue-100 rounded-lg">
                  <Users className="w-5 h-5 text-blue-600" />
                </div>
                <div>
                  <p className="text-2xl font-bold">{accesses.length}</p>
                  <p className="text-sm text-slate-500">Total de acessos</p>
                </div>
              </div>
            </CardContent>
          </Card>
          <Card className="border-0 shadow-sm">
            <CardContent className="p-4">
              <div className="flex items-center gap-3">
                <div className="p-2 bg-emerald-100 rounded-lg">
                  <Key className="w-5 h-5 text-emerald-600" />
                </div>
                <div>
                  <p className="text-2xl font-bold">{accesses.filter(a => a.is_active).length}</p>
                  <p className="text-sm text-slate-500">Ativos</p>
                </div>
              </div>
            </CardContent>
          </Card>
          <Card className="border-0 shadow-sm">
            <CardContent className="p-4">
              <div className="flex items-center gap-3">
                <div className="p-2 bg-amber-100 rounded-lg">
                  <Clock className="w-5 h-5 text-amber-600" />
                </div>
                <div>
                  <p className="text-2xl font-bold">
                    {accesses.filter(a => a.last_access && new Date(a.last_access) > new Date(Date.now() - 7 * 24 * 60 * 60 * 1000)).length}
                  </p>
                  <p className="text-sm text-slate-500">Ativos esta semana</p>
                </div>
              </div>
            </CardContent>
          </Card>
        </div>

        {/* Search */}
        <Card className="border-0 shadow-sm">
          <CardContent className="p-4">
            <div className="relative">
              <Search className="absolute left-3 top-1/2 -translate-y-1/2 w-4 h-4 text-slate-400" />
              <Input
                placeholder="Buscar por cliente..."
                value={searchTerm}
                onChange={(e) => setSearchTerm(e.target.value)}
                className="pl-9"
              />
            </div>
          </CardContent>
        </Card>

        {/* List */}
        <Card className="border-0 shadow-sm">
          {isLoading ? (
            <CardContent className="p-6">
              <div className="animate-pulse space-y-4">
                {[1, 2, 3].map(i => (
                  <div key={i} className="h-20 bg-slate-100 rounded-lg" />
                ))}
              </div>
            </CardContent>
          ) : filteredAccesses.length === 0 ? (
            <CardContent className="p-12 text-center">
              <LinkIcon className="w-12 h-12 mx-auto text-slate-300 mb-4" />
              <p className="text-slate-500">Nenhum acesso criado</p>
              <Button 
                variant="outline" 
                className="mt-4"
                onClick={() => setShowCreateModal(true)}
              >
                <Plus className="w-4 h-4 mr-2" />
                Criar primeiro acesso
              </Button>
            </CardContent>
          ) : (
            <div className="divide-y divide-slate-100">
              {filteredAccesses.map(access => (
                <div key={access.id} className="p-4 hover:bg-slate-50 transition-colors">
                  <div className="flex flex-col md:flex-row md:items-center gap-4">
                    <div className="flex-1">
                      <div className="flex items-center gap-2 mb-1">
                        <span className="font-semibold text-slate-900">{access.customer_name}</span>
                        <Badge className={access.is_active ? 'bg-emerald-100 text-emerald-700' : 'bg-slate-100 text-slate-500'}>
                          {access.is_active ? 'Ativo' : 'Inativo'}
                        </Badge>
                      </div>
                      <div className="text-sm text-slate-500 space-y-1">
                        <p>{access.customer_email} • {access.customer_phone}</p>
                        {access.last_access && (
                          <p className="flex items-center gap-1">
                            <Clock className="w-3 h-3" />
                            Último acesso: {format(new Date(access.last_access), "dd/MM/yyyy 'às' HH:mm", { locale: ptBR })}
                          </p>
                        )}
                      </div>
                    </div>

                    <div className="flex flex-wrap gap-2">
                      <Button
                        variant="outline"
                        size="sm"
                        onClick={() => copyToClipboard(getPortalUrl(access.access_token), access.id)}
                        disabled={!access.is_active}
                      >
                        {copiedId === access.id ? (
                          <>
                            <Check className="w-4 h-4 mr-1 text-emerald-500" />
                            Copiado!
                          </>
                        ) : (
                          <>
                            <Copy className="w-4 h-4 mr-1" />
                            Copiar Link
                          </>
                        )}
                      </Button>
                      
                      <a 
                        href={getPortalUrl(access.access_token)} 
                        target="_blank" 
                        rel="noopener noreferrer"
                      >
                        <Button variant="outline" size="sm" disabled={!access.is_active}>
                          <ExternalLink className="w-4 h-4 mr-1" />
                          Abrir
                        </Button>
                      </a>

                      <Button
                        variant="outline"
                        size="sm"
                        onClick={() => regenerateToken.mutate(access.id)}
                        disabled={regenerateToken.isPending}
                      >
                        <RefreshCw className={`w-4 h-4 mr-1 ${regenerateToken.isPending ? 'animate-spin' : ''}`} />
                        Novo Link
                      </Button>

                      <Button
                        variant="outline"
                        size="sm"
                        onClick={() => toggleAccess.mutate({ id: access.id, is_active: !access.is_active })}
                      >
                        {access.is_active ? (
                          <>
                            <XCircle className="w-4 h-4 mr-1 text-red-500" />
                            Desativar
                          </>
                        ) : (
                          <>
                            <Check className="w-4 h-4 mr-1 text-emerald-500" />
                            Ativar
                          </>
                        )}
                      </Button>

                      <Button
                        variant="outline"
                        size="icon"
                        className="text-red-600 hover:bg-red-50"
                        onClick={() => deleteAccess.mutate(access.id)}
                      >
                        <Trash2 className="w-4 h-4" />
                      </Button>
                    </div>
                  </div>
                </div>
              ))}
            </div>
          )}
        </Card>

        {/* Create Modal */}
        <Dialog open={showCreateModal} onOpenChange={setShowCreateModal}>
          <DialogContent>
            <DialogHeader>
              <DialogTitle>Criar Acesso para Cliente</DialogTitle>
              <DialogDescription>
                Selecione um cliente para gerar seu link de acesso exclusivo
              </DialogDescription>
            </DialogHeader>

            <div className="space-y-4">
              <div className="space-y-2">
                <Label>Cliente</Label>
                <Select value={selectedCustomerId} onValueChange={setSelectedCustomerId}>
                  <SelectTrigger>
                    <SelectValue placeholder="Selecione um cliente" />
                  </SelectTrigger>
                  <SelectContent>
                    {customersWithoutAccess.length === 0 ? (
                      <div className="p-4 text-center text-slate-500">
                        Todos os clientes já possuem acesso
                      </div>
                    ) : (
                      customersWithoutAccess.map(customer => (
                        <SelectItem key={customer.id} value={customer.id}>
                          {customer.name} - {customer.phone}
                        </SelectItem>
                      ))
                    )}
                  </SelectContent>
                </Select>
              </div>

              <div className="flex gap-3 pt-4">
                <Button 
                  variant="outline" 
                  className="flex-1"
                  onClick={() => setShowCreateModal(false)}
                >
                  Cancelar
                </Button>
                <Button 
                  className="flex-1 bg-slate-800 hover:bg-slate-700"
                  onClick={() => createAccess.mutate(selectedCustomerId)}
                  disabled={!selectedCustomerId || createAccess.isPending}
                >
                  {createAccess.isPending ? 'Criando...' : 'Criar Acesso'}
                </Button>
              </div>
            </div>
          </DialogContent>
        </Dialog>
      </div>
    </div>
  );
}
