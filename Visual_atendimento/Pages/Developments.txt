import React, { useState } from 'react';
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { base44 } from '@/api/base44Client';
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Badge } from "@/components/ui/badge";
import { Label } from "@/components/ui/label";
import { Textarea } from "@/components/ui/textarea";
import { Tabs, TabsContent, TabsList, TabsTrigger } from "@/components/ui/tabs";
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
} from "@/components/ui/dialog";
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@/components/ui/select";
import { 
  Search, 
  FileText,
  Clock,
  CheckCircle2,
  XCircle,
  Eye,
  Package,
  Upload,
  X,
  Send,
  Loader2,
  User,
  Calendar,
  DollarSign,
  Image,
  ArrowRightLeft,
  MessageCircle,
  History
} from "lucide-react";
import { format } from "date-fns";
import { ptBR } from "date-fns/locale";
import ArtworkVersionManager from '@/components/developments/ArtworkVersionManager';

const statusConfig = {
  pending: { label: 'Pendente', color: 'bg-amber-100 text-amber-700' },
  in_development: { label: 'Em Desenvolvimento', color: 'bg-blue-100 text-blue-700' },
  awaiting_approval: { label: 'Aguardando Aprovação', color: 'bg-violet-100 text-violet-700' },
  revision_requested: { label: 'Revisão Solicitada', color: 'bg-orange-100 text-orange-700' },
  approved: { label: 'Aprovado', color: 'bg-emerald-100 text-emerald-700' },
  order_created: { label: 'Pedido Emitido', color: 'bg-indigo-100 text-indigo-700' },
  production: { label: 'Em Produção', color: 'bg-cyan-100 text-cyan-700' },
  completed: { label: 'Concluído', color: 'bg-green-100 text-green-700' },
  cancelled: { label: 'Cancelado', color: 'bg-red-100 text-red-700' }
};

export default function Developments() {
  const queryClient = useQueryClient();
  const [searchTerm, setSearchTerm] = useState('');
  const [statusFilter, setStatusFilter] = useState('all');
  const [selectedDev, setSelectedDev] = useState(null);
  const [showDetailModal, setShowDetailModal] = useState(false);
  const [artworkFiles, setArtworkFiles] = useState([]);
  const [unitPrice, setUnitPrice] = useState('');
  const [internalNotes, setInternalNotes] = useState('');
  const [uploading, setUploading] = useState(false);
  const [showTransferModal, setShowTransferModal] = useState(false);
  const [transferTo, setTransferTo] = useState('');
  const [transferReason, setTransferReason] = useState('');
  const [showStartModal, setShowStartModal] = useState(false);
  const [showRevisionModal, setShowRevisionModal] = useState(false);
  const [estimatedDeliveryDate, setEstimatedDeliveryDate] = useState('');

  // Load current user
  const { data: currentUser } = useQuery({
    queryKey: ['currentUser'],
    queryFn: () => base44.auth.me()
  });

  // Load users for transfer
  const { data: users = [] } = useQuery({
    queryKey: ['users'],
    queryFn: () => base44.entities.User.list('full_name', 100)
  });

  // Load developments
  const { data: allDevelopments = [], isLoading } = useQuery({
    queryKey: ['developments'],
    queryFn: () => base44.entities.DevelopmentRequest.list('-created_date', 200),
  });

  // Filter developments by user permissions (admin sees all)
  const isAdmin = currentUser?.role === 'admin';
  const userDepartment = currentUser?.department;
  const developments = isAdmin 
    ? allDevelopments 
    : allDevelopments.filter(d => 
        d.assigned_to === currentUser?.email || 
        d.assigned_role === userDepartment ||
        d.status === 'pending' // Everyone can see pending to pick up
      );

  // Update development
  const updateDev = useMutation({
    mutationFn: async ({ id, data }) => {
      return base44.entities.DevelopmentRequest.update(id, data);
    },
    onSuccess: () => {
      queryClient.invalidateQueries(['developments']);
    }
  });

  // Create notification
  const createNotification = useMutation({
    mutationFn: (data) => base44.entities.Notification.create(data)
  });

  // Transfer development
  const transferDev = useMutation({
    mutationFn: async ({ id, toUser, reason }) => {
      const toUserData = users.find(u => u.email === toUser);
      const transferRecord = {
        from_user: currentUser?.email,
        to_user: toUser,
        from_role: currentUser?.role || 'user',
        to_role: toUserData?.role || 'user',
        reason: reason,
        transferred_at: new Date().toISOString()
      };
      
      const existingHistory = selectedDev.transfer_history || [];
      
      await base44.entities.DevelopmentRequest.update(id, {
        assigned_to: toUser,
        assigned_to_name: toUserData?.full_name,
        transfer_history: [...existingHistory, transferRecord]
      });

      // Log activity
      await base44.entities.ActivityLog.create({
        user_email: currentUser?.email,
        user_name: currentUser?.full_name,
        action_type: 'transfer',
        entity_type: 'development',
        entity_id: id,
        entity_name: selectedDev.request_number,
        description: `Transferido para ${toUserData?.full_name || toUser}. Motivo: ${reason}`
      });

      // Notify recipient
      await createNotification.mutateAsync({
        type: 'development_update',
        title: 'Desenvolvimento transferido para você',
        message: `${selectedDev.request_number} - ${selectedDev.product_name}. Motivo: ${reason}`,
        reference_id: id,
        reference_type: 'development',
        for_user: toUser,
        priority: 'high'
      });
    },
    onSuccess: () => {
      queryClient.invalidateQueries(['developments']);
      setShowTransferModal(false);
      setShowDetailModal(false);
      setTransferTo('');
      setTransferReason('');
    }
  });

  const handleOpenDetail = (dev) => {
    setSelectedDev(dev);
    setArtworkFiles(dev.artwork_files || []);
    setUnitPrice(dev.unit_price?.toString() || '');
    setInternalNotes(dev.internal_notes || '');
    setShowDetailModal(true);
  };

  const handleFileUpload = async (e) => {
    const file = e.target.files?.[0];
    if (!file) return;
    
    setUploading(true);
    const { file_url } = await base44.integrations.Core.UploadFile({ file });
    setArtworkFiles(prev => [...prev, file_url]);
    setUploading(false);
  };

  const removeFile = (index) => {
    setArtworkFiles(prev => prev.filter((_, i) => i !== index));
  };

  const handleUpdateStatus = async (status) => {
    const updateData = {
      status,
      artwork_files: artworkFiles,
      internal_notes: internalNotes
    };

    if (unitPrice) {
      updateData.unit_price = parseFloat(unitPrice);
      updateData.total_price = parseFloat(unitPrice) * selectedDev.quantity;
    }

    if (status === 'in_development' && !selectedDev.assigned_to) {
      updateData.assigned_to = currentUser?.email;
    }

    await updateDev.mutateAsync({ id: selectedDev.id, data: updateData });

    // Send notification to client
    if (status === 'awaiting_approval') {
      await createNotification.mutateAsync({
        type: 'system',
        title: 'Desenvolvimento pronto para aprovação',
        message: `O desenvolvimento do produto "${selectedDev.product_name}" está pronto para sua aprovação.`,
        reference_id: selectedDev.id,
        reference_type: 'order',
        priority: 'high'
      });
    }

    setShowDetailModal(false);
  };

  // Filter developments
  const filteredDevelopments = developments.filter(dev => {
    const matchesSearch = !searchTerm || 
      dev.product_name?.toLowerCase().includes(searchTerm.toLowerCase()) ||
      dev.customer_name?.toLowerCase().includes(searchTerm.toLowerCase()) ||
      dev.request_number?.toLowerCase().includes(searchTerm.toLowerCase());
    
    const matchesStatus = statusFilter === 'all' || dev.status === statusFilter;
    
    return matchesSearch && matchesStatus;
  });

  // Stats
  const stats = {
    pending: developments.filter(d => d.status === 'pending').length,
    in_development: developments.filter(d => d.status === 'in_development').length,
    awaiting_approval: developments.filter(d => d.status === 'awaiting_approval').length,
    revision_requested: developments.filter(d => d.status === 'revision_requested').length,
    approved: developments.filter(d => d.status === 'approved').length
  };

  return (
    <div className="min-h-screen bg-slate-50 p-6">
      <div className="max-w-7xl mx-auto space-y-6">
        {/* Header */}
        <div className="flex flex-col md:flex-row md:items-center md:justify-between gap-4">
          <div>
            <h1 className="text-2xl font-bold text-slate-900">Desenvolvimentos</h1>
            <p className="text-slate-500">Gerencie solicitações de personalização</p>
          </div>
        </div>

        {/* Stats */}
        <div className="grid grid-cols-2 md:grid-cols-5 gap-4">
          <Card className="border-0 shadow-sm cursor-pointer hover:shadow-md transition-shadow" onClick={() => setStatusFilter('pending')}>
            <CardContent className="p-4">
              <div className="flex items-center gap-3">
                <div className="p-2 bg-amber-100 rounded-lg">
                  <Clock className="w-5 h-5 text-amber-600" />
                </div>
                <div>
                  <p className="text-2xl font-bold">{stats.pending}</p>
                  <p className="text-xs text-slate-500">Pendentes</p>
                </div>
              </div>
            </CardContent>
          </Card>
          <Card className="border-0 shadow-sm cursor-pointer hover:shadow-md transition-shadow" onClick={() => setStatusFilter('in_development')}>
            <CardContent className="p-4">
              <div className="flex items-center gap-3">
                <div className="p-2 bg-blue-100 rounded-lg">
                  <Package className="w-5 h-5 text-blue-600" />
                </div>
                <div>
                  <p className="text-2xl font-bold">{stats.in_development}</p>
                  <p className="text-xs text-slate-500">Em Desenv.</p>
                </div>
              </div>
            </CardContent>
          </Card>
          <Card className="border-0 shadow-sm cursor-pointer hover:shadow-md transition-shadow" onClick={() => setStatusFilter('awaiting_approval')}>
            <CardContent className="p-4">
              <div className="flex items-center gap-3">
                <div className="p-2 bg-violet-100 rounded-lg">
                  <Eye className="w-5 h-5 text-violet-600" />
                </div>
                <div>
                  <p className="text-2xl font-bold">{stats.awaiting_approval}</p>
                  <p className="text-xs text-slate-500">Aguard. Aprov.</p>
                </div>
              </div>
            </CardContent>
          </Card>
          <Card className="border-0 shadow-sm cursor-pointer hover:shadow-md transition-shadow" onClick={() => setStatusFilter('revision_requested')}>
            <CardContent className="p-4">
              <div className="flex items-center gap-3">
                <div className="p-2 bg-orange-100 rounded-lg">
                  <FileText className="w-5 h-5 text-orange-600" />
                </div>
                <div>
                  <p className="text-2xl font-bold">{stats.revision_requested}</p>
                  <p className="text-xs text-slate-500">Revisão</p>
                </div>
              </div>
            </CardContent>
          </Card>
          <Card className="border-0 shadow-sm cursor-pointer hover:shadow-md transition-shadow" onClick={() => setStatusFilter('approved')}>
            <CardContent className="p-4">
              <div className="flex items-center gap-3">
                <div className="p-2 bg-emerald-100 rounded-lg">
                  <CheckCircle2 className="w-5 h-5 text-emerald-600" />
                </div>
                <div>
                  <p className="text-2xl font-bold">{stats.approved}</p>
                  <p className="text-xs text-slate-500">Aprovados</p>
                </div>
              </div>
            </CardContent>
          </Card>
        </div>

        {/* Filters */}
        <Card className="border-0 shadow-sm">
          <CardContent className="p-4">
            <div className="flex flex-col md:flex-row gap-4">
              <div className="relative flex-1">
                <Search className="absolute left-3 top-1/2 -translate-y-1/2 w-4 h-4 text-slate-400" />
                <Input
                  placeholder="Buscar por produto, cliente ou número..."
                  value={searchTerm}
                  onChange={(e) => setSearchTerm(e.target.value)}
                  className="pl-9"
                />
              </div>
              
              <Select value={statusFilter} onValueChange={setStatusFilter}>
                <SelectTrigger className="w-full md:w-48">
                  <SelectValue placeholder="Status" />
                </SelectTrigger>
                <SelectContent>
                  <SelectItem value="all">Todos os status</SelectItem>
                  {Object.entries(statusConfig).map(([key, config]) => (
                    <SelectItem key={key} value={key}>{config.label}</SelectItem>
                  ))}
                </SelectContent>
              </Select>
            </div>
          </CardContent>
        </Card>

        {/* Developments List */}
        <Card className="border-0 shadow-sm">
          {isLoading ? (
            <CardContent className="p-6">
              <div className="animate-pulse space-y-4">
                {[1, 2, 3, 4, 5].map(i => (
                  <div key={i} className="h-20 bg-slate-100 rounded-lg" />
                ))}
              </div>
            </CardContent>
          ) : filteredDevelopments.length === 0 ? (
            <CardContent className="p-12 text-center">
              <FileText className="w-12 h-12 mx-auto text-slate-300 mb-4" />
              <p className="text-slate-500">Nenhum desenvolvimento encontrado</p>
            </CardContent>
          ) : (
            <div className="divide-y divide-slate-100">
              {filteredDevelopments.map(dev => (
                <div 
                  key={dev.id} 
                  className="p-4 hover:bg-slate-50 cursor-pointer transition-colors"
                  onClick={() => handleOpenDetail(dev)}
                >
                  <div className="flex items-center gap-4">
                    <div className={`w-12 h-12 rounded-xl flex items-center justify-center ${statusConfig[dev.status]?.color.split(' ')[0]}`}>
                      <Package className="w-6 h-6" />
                    </div>
                    
                    <div className="flex-1 min-w-0">
                      <div className="flex items-center gap-2 mb-1">
                        <span className="font-semibold text-slate-900">{dev.product_name}</span>
                        <Badge variant="outline" className="text-xs">
                          #{dev.request_number}
                        </Badge>
                        <Badge className={statusConfig[dev.status]?.color}>
                          {statusConfig[dev.status]?.label}
                        </Badge>
                      </div>
                      <div className="flex items-center gap-4 text-sm text-slate-500">
                        <span className="flex items-center gap-1">
                          <User className="w-3 h-3" />
                          {dev.customer_name}
                        </span>
                        <span>{dev.quantity} un.</span>
                        <span>{format(new Date(dev.created_date), "dd/MM/yyyy")}</span>
                      </div>
                    </div>

                    {dev.total_price && (
                      <div className="text-right hidden md:block">
                        <p className="font-semibold text-slate-900">
                          R$ {dev.total_price.toFixed(2)}
                        </p>
                      </div>
                    )}
                  </div>
                </div>
              ))}
            </div>
          )}
        </Card>

        {/* Detail Modal */}
        <Dialog open={showDetailModal} onOpenChange={setShowDetailModal}>
          <DialogContent className="max-w-3xl max-h-[90vh] overflow-y-auto">
            <DialogHeader>
              <DialogTitle>Gerenciar Desenvolvimento</DialogTitle>
            </DialogHeader>

            {selectedDev && (
              <div className="space-y-6">
                {/* Header */}
                <div className="flex items-center justify-between p-4 bg-slate-50 rounded-xl">
                  <div>
                    <p className="text-sm text-slate-500">#{selectedDev.request_number}</p>
                    <h3 className="text-xl font-semibold text-slate-900">{selectedDev.product_name}</h3>
                    <p className="text-sm text-slate-500">{selectedDev.customer_name}</p>
                  </div>
                  <Badge className={statusConfig[selectedDev.status]?.color}>
                    {statusConfig[selectedDev.status]?.label}
                  </Badge>
                </div>

                {/* Info */}
                <div className="grid grid-cols-2 md:grid-cols-4 gap-4">
                  <div className="p-3 bg-slate-50 rounded-lg">
                    <p className="text-xs text-slate-500">Quantidade</p>
                    <p className="font-semibold">{selectedDev.quantity} un.</p>
                    <p className="text-xs text-slate-400">Mín: {selectedDev.min_order_quantity || 100}</p>
                  </div>
                  <div className="p-3 bg-slate-50 rounded-lg">
                    <p className="text-xs text-slate-500">Preço Base</p>
                    <p className="font-semibold">R$ {selectedDev.product_base_price?.toFixed(2) || '0,00'}</p>
                  </div>
                  <div className="p-3 bg-slate-50 rounded-lg">
                    <p className="text-xs text-slate-500">Solicitado em</p>
                    <p className="font-semibold">{format(new Date(selectedDev.created_date), "dd/MM/yyyy")}</p>
                  </div>
                  {selectedDev.estimated_delivery_date && (
                    <div className="p-3 bg-blue-50 rounded-lg">
                      <p className="text-xs text-blue-600">Previsão Entrega</p>
                      <p className="font-semibold text-blue-900">{format(new Date(selectedDev.estimated_delivery_date), "dd/MM/yyyy")}</p>
                    </div>
                  )}
                  {selectedDev.assigned_to && (
                    <div className="p-3 bg-slate-50 rounded-lg">
                      <p className="text-xs text-slate-500">Responsável</p>
                      <p className="font-semibold text-sm">{selectedDev.assigned_to_name || selectedDev.assigned_to}</p>
                    </div>
                  )}
                </div>

                {/* Selected Processes */}
                {selectedDev.selected_processes?.length > 0 && (
                  <div>
                    <h4 className="font-semibold text-slate-900 mb-2">Processos Selecionados</h4>
                    <div className="flex flex-wrap gap-2">
                      {selectedDev.selected_processes.map((proc, idx) => (
                        <Badge key={idx} className="bg-violet-100 text-violet-700">
                          {proc.process_name} - R$ {proc.price_per_unit?.toFixed(2)}/un
                        </Badge>
                      ))}
                    </div>
                  </div>
                )}

                {/* Selected Materials */}
                {selectedDev.selected_materials?.length > 0 && (
                  <div>
                    <h4 className="font-semibold text-slate-900 mb-2">Materiais Selecionados</h4>
                    <div className="space-y-2">
                      {selectedDev.selected_materials.map((mat, idx) => (
                        <div key={idx} className="flex items-center gap-3 p-2 bg-amber-50 rounded-lg">
                          <span className="font-medium">{mat.display_name || mat.material_name}</span>
                          {mat.selected_color && (
                            <Badge className="bg-white border">Cor: {mat.selected_color}</Badge>
                          )}
                        </div>
                      ))}
                    </div>
                  </div>
                )}

                {/* Customizations */}
                {selectedDev.customizations?.length > 0 && (
                  <div>
                    <h4 className="font-semibold text-slate-900 mb-2">Personalizações</h4>
                    <div className="flex flex-wrap gap-2">
                      {selectedDev.customizations.map((cust, idx) => (
                        <Badge key={idx} variant="outline">
                          {cust.name}: {cust.selected_label}
                        </Badge>
                      ))}
                    </div>
                  </div>
                )}

                {/* Description */}
                {selectedDev.description && (
                  <div>
                    <h4 className="font-semibold text-slate-900 mb-2">Descrição do Cliente</h4>
                    <p className="text-slate-600 whitespace-pre-wrap p-3 bg-slate-50 rounded-lg">
                      {selectedDev.description}
                    </p>
                  </div>
                )}

                {/* Reference Files */}
                {selectedDev.reference_files?.length > 0 && (
                  <div>
                    <h4 className="font-semibold text-slate-900 mb-2">Arquivos de Referência</h4>
                    <div className="flex flex-wrap gap-3">
                      {selectedDev.reference_files.map((url, idx) => (
                        <a 
                          key={idx}
                          href={url}
                          target="_blank"
                          rel="noopener noreferrer"
                          className="w-24 h-24 rounded-lg overflow-hidden border hover:border-slate-400"
                        >
                          <img src={url} alt="" className="w-full h-full object-cover" />
                        </a>
                      ))}
                    </div>
                  </div>
                )}

                {/* Revision Notes */}
                {selectedDev.revision_notes && (
                  <div className="p-4 bg-orange-50 rounded-xl border border-orange-200">
                    <h4 className="font-semibold text-orange-900 mb-2">Notas de Revisão do Cliente</h4>
                    <p className="text-orange-800">{selectedDev.revision_notes}</p>
                  </div>
                )}

                {/* Artwork Version Manager */}
                <ArtworkVersionManager 
                  developmentId={selectedDev.id}
                  currentUser={currentUser}
                  onVersionAdded={() => queryClient.invalidateQueries(['developments'])}
                />

                {/* Pricing */}
                <div className="grid grid-cols-2 gap-4">
                  <div className="space-y-2">
                    <Label>Preço Unitário (R$)</Label>
                    <Input
                      type="number"
                      step="0.01"
                      value={unitPrice}
                      onChange={(e) => setUnitPrice(e.target.value)}
                      placeholder="0,00"
                    />
                  </div>
                  <div className="space-y-2">
                    <Label>Valor Total</Label>
                    <div className="h-10 px-3 flex items-center bg-slate-100 rounded-md font-semibold">
                      R$ {unitPrice ? (parseFloat(unitPrice) * selectedDev.quantity).toFixed(2) : '0,00'}
                    </div>
                  </div>
                </div>

                {/* Internal Notes */}
                <div className="space-y-2">
                  <Label>Notas Internas</Label>
                  <Textarea
                    value={internalNotes}
                    onChange={(e) => setInternalNotes(e.target.value)}
                    placeholder="Anotações internas da equipe..."
                    rows={3}
                  />
                </div>

                {/* Actions */}
                <div className="flex flex-wrap gap-3 pt-4 border-t">
                  {selectedDev.status === 'pending' && (
                    <Button 
                      onClick={() => setShowStartModal(true)}
                      className="bg-blue-600 hover:bg-blue-700"
                      disabled={updateDev.isPending}
                    >
                      <Package className="w-4 h-4 mr-2" />
                      Iniciar Desenvolvimento
                    </Button>
                  )}
                  
                  {selectedDev.status === 'in_development' && (
                    <Button 
                      onClick={() => handleUpdateStatus('awaiting_approval')}
                      className="bg-violet-600 hover:bg-violet-700"
                      disabled={updateDev.isPending || !unitPrice}
                    >
                      <Send className="w-4 h-4 mr-2" />
                      Enviar para Aprovação
                    </Button>
                  )}

                  {selectedDev.status === 'revision_requested' && (
                    <Button 
                      onClick={() => setShowRevisionModal(true)}
                      className="bg-orange-600 hover:bg-orange-700"
                      disabled={updateDev.isPending}
                    >
                      <Package className="w-4 h-4 mr-2" />
                      Iniciar Revisão
                    </Button>
                  )}

                  {selectedDev.status === 'approved' && (
                    <Button 
                      onClick={() => handleUpdateStatus('production')}
                      className="bg-cyan-600 hover:bg-cyan-700"
                      disabled={updateDev.isPending}
                    >
                      <Package className="w-4 h-4 mr-2" />
                      Iniciar Produção
                    </Button>
                  )}

                  {selectedDev.status === 'production' && (
                    <Button 
                      onClick={() => handleUpdateStatus('completed')}
                      className="bg-green-600 hover:bg-green-700"
                      disabled={updateDev.isPending}
                    >
                      <CheckCircle2 className="w-4 h-4 mr-2" />
                      Finalizar
                    </Button>
                  )}

                  <Button 
                    variant="outline"
                    onClick={() => {
                      updateDev.mutate({ 
                        id: selectedDev.id, 
                        data: { artwork_files: artworkFiles, internal_notes: internalNotes, unit_price: parseFloat(unitPrice) || undefined, total_price: unitPrice ? parseFloat(unitPrice) * selectedDev.quantity : undefined }
                      });
                      setShowDetailModal(false);
                    }}
                    disabled={updateDev.isPending}
                  >
                    Salvar Alterações
                  </Button>

                  <Button 
                    variant="outline"
                    onClick={() => setShowTransferModal(true)}
                    disabled={updateDev.isPending}
                  >
                    <ArrowRightLeft className="w-4 h-4 mr-2" />
                    Transferir
                  </Button>

                  {selectedDev.status !== 'cancelled' && selectedDev.status !== 'completed' && (
                    <Button 
                      variant="outline"
                      className="text-red-600 border-red-200 hover:bg-red-50"
                      onClick={() => handleUpdateStatus('cancelled')}
                      disabled={updateDev.isPending}
                    >
                      <XCircle className="w-4 h-4 mr-2" />
                      Cancelar
                    </Button>
                  )}
                </div>

                {/* Transfer History */}
                {selectedDev.transfer_history?.length > 0 && (
                  <div className="mt-6 pt-4 border-t">
                    <h4 className="font-semibold text-slate-900 flex items-center gap-2 mb-3">
                      <History className="w-4 h-4" />
                      Histórico de Transferências
                    </h4>
                    <div className="space-y-2">
                      {selectedDev.transfer_history.map((transfer, idx) => (
                        <div key={idx} className="text-sm p-3 bg-slate-50 rounded-lg">
                          <p className="text-slate-700">
                            <strong>{transfer.from_user}</strong> → <strong>{transfer.to_user}</strong>
                          </p>
                          <p className="text-slate-500">{transfer.reason}</p>
                          <p className="text-xs text-slate-400 mt-1">
                            {format(new Date(transfer.transferred_at), "dd/MM/yyyy 'às' HH:mm", { locale: ptBR })}
                          </p>
                        </div>
                      ))}
                    </div>
                  </div>
                )}
              </div>
            )}
          </DialogContent>
        </Dialog>

        {/* Revision Modal */}
        <Dialog open={showRevisionModal} onOpenChange={setShowRevisionModal}>
          <DialogContent>
            <DialogHeader>
              <DialogTitle>Iniciar Revisão</DialogTitle>
            </DialogHeader>
            <div className="space-y-4">
              <div className="p-4 bg-orange-50 rounded-xl border border-orange-200">
                <p className="font-medium text-orange-900">{selectedDev?.product_name}</p>
                <p className="text-sm text-orange-700">
                  {selectedDev?.customer_name} • {selectedDev?.quantity} un.
                </p>
                {selectedDev?.revision_notes && (
                  <div className="mt-3 pt-3 border-t border-orange-200">
                    <p className="text-xs text-orange-600 font-medium mb-1">Solicitação do cliente:</p>
                    <p className="text-sm text-orange-800">{selectedDev.revision_notes}</p>
                  </div>
                )}
              </div>

              <div className="space-y-2">
                <Label>Nova Data Prevista de Entrega *</Label>
                <Input
                  type="date"
                  value={estimatedDeliveryDate}
                  onChange={(e) => setEstimatedDeliveryDate(e.target.value)}
                  min={format(new Date(), 'yyyy-MM-dd')}
                  required
                />
                <p className="text-xs text-slate-500">
                  Informe a nova previsão de entrega após a revisão
                </p>
              </div>

              <div className="flex gap-3 pt-4">
                <Button variant="outline" className="flex-1" onClick={() => setShowRevisionModal(false)}>
                  Cancelar
                </Button>
                <Button 
                  className="flex-1 bg-orange-600 hover:bg-orange-700"
                  onClick={async () => {
                    await updateDev.mutateAsync({
                      id: selectedDev.id,
                      data: {
                        status: 'in_development',
                        estimated_delivery_date: estimatedDeliveryDate,
                        assigned_to: currentUser?.email,
                        assigned_to_name: currentUser?.full_name
                      }
                    });
                    setShowRevisionModal(false);
                    setShowDetailModal(false);
                    setEstimatedDeliveryDate('');
                  }}
                  disabled={!estimatedDeliveryDate || updateDev.isPending}
                >
                  {updateDev.isPending ? 'Iniciando...' : 'Iniciar Revisão'}
                </Button>
              </div>
            </div>
          </DialogContent>
        </Dialog>

        {/* Start Development Modal */}
        <Dialog open={showStartModal} onOpenChange={setShowStartModal}>
          <DialogContent>
            <DialogHeader>
              <DialogTitle>Iniciar Desenvolvimento</DialogTitle>
            </DialogHeader>
            <div className="space-y-4">
              <div className="p-4 bg-slate-50 rounded-xl">
                <p className="font-medium">{selectedDev?.product_name}</p>
                <p className="text-sm text-slate-500">
                  {selectedDev?.customer_name} • {selectedDev?.quantity} un.
                </p>
              </div>

              <div className="space-y-2">
                <Label>Data Prevista de Entrega *</Label>
                <Input
                  type="date"
                  value={estimatedDeliveryDate}
                  onChange={(e) => setEstimatedDeliveryDate(e.target.value)}
                  min={format(new Date(), 'yyyy-MM-dd')}
                  required
                />
                <p className="text-xs text-slate-500">
                  Informe a previsão de entrega para o cliente
                </p>
              </div>

              <div className="flex gap-3 pt-4">
                <Button variant="outline" className="flex-1" onClick={() => setShowStartModal(false)}>
                  Cancelar
                </Button>
                <Button 
                  className="flex-1 bg-blue-600 hover:bg-blue-700"
                  onClick={async () => {
                    await updateDev.mutateAsync({
                      id: selectedDev.id,
                      data: {
                        status: 'in_development',
                        estimated_delivery_date: estimatedDeliveryDate,
                        assigned_to: currentUser?.email,
                        assigned_to_name: currentUser?.full_name
                      }
                    });
                    setShowStartModal(false);
                    setShowDetailModal(false);
                    setEstimatedDeliveryDate('');
                  }}
                  disabled={!estimatedDeliveryDate || updateDev.isPending}
                >
                  {updateDev.isPending ? 'Iniciando...' : 'Iniciar Desenvolvimento'}
                </Button>
              </div>
            </div>
          </DialogContent>
        </Dialog>

        {/* Transfer Modal */}
        <Dialog open={showTransferModal} onOpenChange={setShowTransferModal}>
          <DialogContent>
            <DialogHeader>
              <DialogTitle>Transferir Desenvolvimento</DialogTitle>
            </DialogHeader>
            <div className="space-y-4">
              <div className="space-y-2">
                <Label>Transferir para</Label>
                <Select value={transferTo} onValueChange={setTransferTo}>
                  <SelectTrigger>
                    <SelectValue placeholder="Selecione o destinatário" />
                  </SelectTrigger>
                  <SelectContent>
                    {users.filter(u => u.email !== currentUser?.email).map(user => (
                      <SelectItem key={user.id} value={user.email}>
                        {user.full_name} ({user.role})
                      </SelectItem>
                    ))}
                  </SelectContent>
                </Select>
              </div>

              <div className="space-y-2">
                <Label>Motivo da transferência</Label>
                <Textarea
                  value={transferReason}
                  onChange={(e) => setTransferReason(e.target.value)}
                  placeholder="Descreva o motivo da transferência..."
                  rows={3}
                />
              </div>

              <div className="flex gap-3 pt-4">
                <Button variant="outline" className="flex-1" onClick={() => setShowTransferModal(false)}>
                  Cancelar
                </Button>
                <Button 
                  className="flex-1"
                  onClick={() => transferDev.mutate({ id: selectedDev.id, toUser: transferTo, reason: transferReason })}
                  disabled={!transferTo || !transferReason || transferDev.isPending}
                >
                  {transferDev.isPending ? 'Transferindo...' : 'Confirmar Transferência'}
                </Button>
              </div>
            </div>
          </DialogContent>
        </Dialog>
      </div>
    </div>
  );
}
