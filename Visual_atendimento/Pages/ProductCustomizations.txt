import React, { useState } from 'react';
import { useQuery, useMutation, useQueryClient } from '@tanstack/react-query';
import { base44 } from '@/api/base44Client';
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Badge } from "@/components/ui/badge";
import { Label } from "@/components/ui/label";
import { Textarea } from "@/components/ui/textarea";
import { Switch } from "@/components/ui/switch";
import {
  Dialog,
  DialogContent,
  DialogHeader,
  DialogTitle,
} from "@/components/ui/dialog";
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@/components/ui/select";
import { 
  Search, 
  Plus,
  Settings,
  Palette,
  Layers,
  Ruler,
  Edit,
  Trash2,
  Package,
  X,
  GripVertical
} from "lucide-react";

const typeConfig = {
  color: { label: 'Cor', icon: Palette, color: 'bg-pink-100 text-pink-700' },
  material: { label: 'Material', icon: Layers, color: 'bg-amber-100 text-amber-700' },
  technique: { label: 'Técnica', icon: Settings, color: 'bg-blue-100 text-blue-700' },
  print_area: { label: 'Área de Impressão', icon: Ruler, color: 'bg-green-100 text-green-700' },
  size: { label: 'Tamanho', icon: Ruler, color: 'bg-violet-100 text-violet-700' },
  other: { label: 'Outro', icon: Package, color: 'bg-slate-100 text-slate-700' }
};

export default function ProductCustomizations() {
  const queryClient = useQueryClient();
  const [selectedProductId, setSelectedProductId] = useState('all');
  const [showModal, setShowModal] = useState(false);
  const [editingCustomization, setEditingCustomization] = useState(null);
  const [formData, setFormData] = useState({
    product_id: '',
    product_name: '',
    customization_type: 'color',
    name: '',
    options: [],
    is_required: false,
    min_quantity: '',
    notes: '',
    order: 0
  });
  const [newOption, setNewOption] = useState({ value: '', label: '', price_adjustment: '' });

  // Load products
  const { data: products = [] } = useQuery({
    queryKey: ['products'],
    queryFn: () => base44.entities.Product.list('name', 500),
  });

  // Load customizations
  const { data: customizations = [], isLoading } = useQuery({
    queryKey: ['customizations'],
    queryFn: () => base44.entities.ProductCustomization.list('order', 500),
  });

  // Create/Update customization
  const saveMutation = useMutation({
    mutationFn: async (data) => {
      const payload = {
        ...data,
        min_quantity: data.min_quantity ? parseInt(data.min_quantity) : null,
        order: data.order || 0
      };
      
      if (editingCustomization) {
        return base44.entities.ProductCustomization.update(editingCustomization.id, payload);
      }
      return base44.entities.ProductCustomization.create(payload);
    },
    onSuccess: () => {
      queryClient.invalidateQueries(['customizations']);
      handleCloseModal();
    }
  });

  // Delete customization
  const deleteMutation = useMutation({
    mutationFn: (id) => base44.entities.ProductCustomization.delete(id),
    onSuccess: () => {
      queryClient.invalidateQueries(['customizations']);
    }
  });

  const handleOpenModal = (customization = null, productId = null) => {
    if (customization) {
      setEditingCustomization(customization);
      setFormData({
        product_id: customization.product_id,
        product_name: customization.product_name,
        customization_type: customization.customization_type,
        name: customization.name,
        options: customization.options || [],
        is_required: customization.is_required || false,
        min_quantity: customization.min_quantity?.toString() || '',
        notes: customization.notes || '',
        order: customization.order || 0
      });
    } else {
      setEditingCustomization(null);
      const product = productId ? products.find(p => p.id === productId) : null;
      setFormData({
        product_id: productId || '',
        product_name: product?.name || '',
        customization_type: 'color',
        name: '',
        options: [],
        is_required: false,
        min_quantity: '',
        notes: '',
        order: 0
      });
    }
    setNewOption({ value: '', label: '', price_adjustment: '' });
    setShowModal(true);
  };

  const handleCloseModal = () => {
    setShowModal(false);
    setEditingCustomization(null);
  };

  const addOption = () => {
    if (!newOption.value || !newOption.label) return;
    setFormData(prev => ({
      ...prev,
      options: [...prev.options, {
        value: newOption.value,
        label: newOption.label,
        price_adjustment: newOption.price_adjustment ? parseFloat(newOption.price_adjustment) : 0
      }]
    }));
    setNewOption({ value: '', label: '', price_adjustment: '' });
  };

  const removeOption = (index) => {
    setFormData(prev => ({
      ...prev,
      options: prev.options.filter((_, i) => i !== index)
    }));
  };

  const handleProductChange = (productId) => {
    const product = products.find(p => p.id === productId);
    setFormData(prev => ({
      ...prev,
      product_id: productId,
      product_name: product?.name || ''
    }));
  };

  const handleSubmit = (e) => {
    e.preventDefault();
    saveMutation.mutate(formData);
  };

  // Filter customizations
  const filteredCustomizations = customizations.filter(c => 
    selectedProductId === 'all' || c.product_id === selectedProductId
  );

  // Group by product
  const groupedCustomizations = filteredCustomizations.reduce((acc, c) => {
    if (!acc[c.product_id]) {
      acc[c.product_id] = {
        product_name: c.product_name,
        items: []
      };
    }
    acc[c.product_id].items.push(c);
    return acc;
  }, {});

  return (
    <div className="min-h-screen bg-slate-50 p-6">
      <div className="max-w-7xl mx-auto space-y-6">
        {/* Header */}
        <div className="flex flex-col md:flex-row md:items-center md:justify-between gap-4">
          <div>
            <h1 className="text-2xl font-bold text-slate-900">Personalizações de Produtos</h1>
            <p className="text-slate-500">Configure opções de personalização para cada produto</p>
          </div>
          <Button onClick={() => handleOpenModal()} className="bg-slate-800 hover:bg-slate-700">
            <Plus className="w-4 h-4 mr-2" />
            Nova Personalização
          </Button>
        </div>

        {/* Filter */}
        <Card className="border-0 shadow-sm">
          <CardContent className="p-4">
            <div className="flex items-center gap-4">
              <Label>Filtrar por produto:</Label>
              <Select value={selectedProductId} onValueChange={setSelectedProductId}>
                <SelectTrigger className="w-64">
                  <SelectValue placeholder="Todos os produtos" />
                </SelectTrigger>
                <SelectContent>
                  <SelectItem value="all">Todos os produtos</SelectItem>
                  {products.map(product => (
                    <SelectItem key={product.id} value={product.id}>
                      {product.name}
                    </SelectItem>
                  ))}
                </SelectContent>
              </Select>
            </div>
          </CardContent>
        </Card>

        {/* Customizations List */}
        {isLoading ? (
          <Card className="border-0 shadow-sm">
            <CardContent className="p-6">
              <div className="animate-pulse space-y-4">
                {[1, 2, 3].map(i => (
                  <div key={i} className="h-20 bg-slate-100 rounded-lg" />
                ))}
              </div>
            </CardContent>
          </Card>
        ) : Object.keys(groupedCustomizations).length === 0 ? (
          <Card className="border-0 shadow-sm">
            <CardContent className="p-12 text-center">
              <Settings className="w-12 h-12 mx-auto text-slate-300 mb-4" />
              <p className="text-slate-500">Nenhuma personalização configurada</p>
              <Button 
                variant="outline" 
                className="mt-4"
                onClick={() => handleOpenModal()}
              >
                <Plus className="w-4 h-4 mr-2" />
                Criar personalização
              </Button>
            </CardContent>
          </Card>
        ) : (
          <div className="space-y-6">
            {Object.entries(groupedCustomizations).map(([productId, group]) => (
              <Card key={productId} className="border-0 shadow-sm">
                <CardHeader className="flex flex-row items-center justify-between pb-2">
                  <CardTitle className="text-lg flex items-center gap-2">
                    <Package className="w-5 h-5 text-slate-400" />
                    {group.product_name}
                  </CardTitle>
                  <Button 
                    variant="outline" 
                    size="sm"
                    onClick={() => handleOpenModal(null, productId)}
                  >
                    <Plus className="w-4 h-4 mr-1" />
                    Adicionar
                  </Button>
                </CardHeader>
                <CardContent className="pt-0">
                  <div className="divide-y divide-slate-100">
                    {group.items.map(cust => {
                      const TypeIcon = typeConfig[cust.customization_type]?.icon || Settings;
                      return (
                        <div key={cust.id} className="py-4 first:pt-0 last:pb-0">
                          <div className="flex items-start gap-4">
                            <div className={`w-10 h-10 rounded-lg flex items-center justify-center ${typeConfig[cust.customization_type]?.color.split(' ')[0]}`}>
                              <TypeIcon className="w-5 h-5" />
                            </div>
                            
                            <div className="flex-1">
                              <div className="flex items-center gap-2 mb-1">
                                <span className="font-medium text-slate-900">{cust.name}</span>
                                <Badge className={typeConfig[cust.customization_type]?.color}>
                                  {typeConfig[cust.customization_type]?.label}
                                </Badge>
                                {cust.is_required && (
                                  <Badge variant="outline" className="text-xs">Obrigatório</Badge>
                                )}
                              </div>
                              
                              <div className="flex flex-wrap gap-1 mt-2">
                                {cust.options?.slice(0, 5).map((opt, idx) => (
                                  <Badge key={idx} variant="outline" className="text-xs">
                                    {opt.label}
                                    {opt.price_adjustment > 0 && ` (+R$${opt.price_adjustment})`}
                                  </Badge>
                                ))}
                                {cust.options?.length > 5 && (
                                  <Badge variant="outline" className="text-xs">
                                    +{cust.options.length - 5} mais
                                  </Badge>
                                )}
                              </div>
                              
                              {cust.notes && (
                                <p className="text-sm text-slate-500 mt-2">{cust.notes}</p>
                              )}
                            </div>

                            <div className="flex gap-2">
                              <Button 
                                variant="ghost" 
                                size="icon"
                                onClick={() => handleOpenModal(cust)}
                              >
                                <Edit className="w-4 h-4" />
                              </Button>
                              <Button 
                                variant="ghost" 
                                size="icon"
                                className="text-red-600 hover:text-red-700"
                                onClick={() => deleteMutation.mutate(cust.id)}
                              >
                                <Trash2 className="w-4 h-4" />
                              </Button>
                            </div>
                          </div>
                        </div>
                      );
                    })}
                  </div>
                </CardContent>
              </Card>
            ))}
          </div>
        )}

        {/* Modal */}
        <Dialog open={showModal} onOpenChange={setShowModal}>
          <DialogContent className="max-w-2xl max-h-[90vh] overflow-y-auto">
            <DialogHeader>
              <DialogTitle>
                {editingCustomization ? 'Editar Personalização' : 'Nova Personalização'}
              </DialogTitle>
            </DialogHeader>

            <form onSubmit={handleSubmit} className="space-y-6">
              {/* Product */}
              <div className="space-y-2">
                <Label>Produto *</Label>
                <Select 
                  value={formData.product_id} 
                  onValueChange={handleProductChange}
                  required
                >
                  <SelectTrigger>
                    <SelectValue placeholder="Selecione o produto" />
                  </SelectTrigger>
                  <SelectContent>
                    {products.map(product => (
                      <SelectItem key={product.id} value={product.id}>
                        {product.name}
                      </SelectItem>
                    ))}
                  </SelectContent>
                </Select>
              </div>

              {/* Type and Name */}
              <div className="grid grid-cols-2 gap-4">
                <div className="space-y-2">
                  <Label>Tipo *</Label>
                  <Select 
                    value={formData.customization_type} 
                    onValueChange={(v) => setFormData({ ...formData, customization_type: v })}
                  >
                    <SelectTrigger>
                      <SelectValue />
                    </SelectTrigger>
                    <SelectContent>
                      {Object.entries(typeConfig).map(([key, config]) => (
                        <SelectItem key={key} value={key}>
                          {config.label}
                        </SelectItem>
                      ))}
                    </SelectContent>
                  </Select>
                </div>
                <div className="space-y-2">
                  <Label>Nome da Personalização *</Label>
                  <Input
                    value={formData.name}
                    onChange={(e) => setFormData({ ...formData, name: e.target.value })}
                    placeholder="Ex: Cores disponíveis"
                    required
                  />
                </div>
              </div>

              {/* Options */}
              <div className="space-y-3">
                <Label>Opções</Label>
                
                {formData.options.length > 0 && (
                  <div className="space-y-2">
                    {formData.options.map((opt, idx) => (
                      <div key={idx} className="flex items-center gap-2 p-2 bg-slate-50 rounded-lg">
                        <span className="flex-1 text-sm">
                          <strong>{opt.label}</strong> ({opt.value})
                          {opt.price_adjustment > 0 && (
                            <span className="text-slate-500"> +R$ {opt.price_adjustment.toFixed(2)}</span>
                          )}
                        </span>
                        <Button
                          type="button"
                          variant="ghost"
                          size="icon"
                          onClick={() => removeOption(idx)}
                        >
                          <X className="w-4 h-4" />
                        </Button>
                      </div>
                    ))}
                  </div>
                )}

                <div className="flex gap-2">
                  <Input
                    placeholder="Valor (interno)"
                    value={newOption.value}
                    onChange={(e) => setNewOption({ ...newOption, value: e.target.value })}
                    className="flex-1"
                  />
                  <Input
                    placeholder="Label (exibido)"
                    value={newOption.label}
                    onChange={(e) => setNewOption({ ...newOption, label: e.target.value })}
                    className="flex-1"
                  />
                  <Input
                    type="number"
                    step="0.01"
                    placeholder="Ajuste R$"
                    value={newOption.price_adjustment}
                    onChange={(e) => setNewOption({ ...newOption, price_adjustment: e.target.value })}
                    className="w-28"
                  />
                  <Button type="button" variant="outline" onClick={addOption}>
                    <Plus className="w-4 h-4" />
                  </Button>
                </div>
              </div>

              {/* Settings */}
              <div className="grid grid-cols-2 gap-4">
                <div className="flex items-center gap-3">
                  <Switch
                    checked={formData.is_required}
                    onCheckedChange={(v) => setFormData({ ...formData, is_required: v })}
                  />
                  <Label>Obrigatório</Label>
                </div>
                <div className="space-y-2">
                  <Label>Quantidade Mínima</Label>
                  <Input
                    type="number"
                    value={formData.min_quantity}
                    onChange={(e) => setFormData({ ...formData, min_quantity: e.target.value })}
                    placeholder="0"
                  />
                </div>
              </div>

              {/* Notes */}
              <div className="space-y-2">
                <Label>Observações</Label>
                <Textarea
                  value={formData.notes}
                  onChange={(e) => setFormData({ ...formData, notes: e.target.value })}
                  placeholder="Informações adicionais sobre esta personalização..."
                  rows={2}
                />
              </div>

              {/* Actions */}
              <div className="flex justify-end gap-3 pt-4 border-t">
                <Button type="button" variant="outline" onClick={handleCloseModal}>
                  Cancelar
                </Button>
                <Button type="submit" disabled={saveMutation.isPending}>
                  {saveMutation.isPending ? 'Salvando...' : 'Salvar'}
                </Button>
              </div>
            </form>
          </DialogContent>
        </Dialog>
      </div>
    </div>
  );
}
